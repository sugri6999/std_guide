									var ui = {
										init : function(){
											if ($('.tab-nav').length){this.tab.init();} // Tab
										},

										/*
											기능정의 : #Tab
											연결방식 : href="" / id=""
											화면경로 : modules/modules_tab.html
											화면메뉴 : 대메뉴 > 중메뉴 > 소메뉴 > 화면명
										*/
										tab : {
											eleButton : '.tab-nav a',
											eleContent : '.tab-content',
											clsActive : 'is-active',
											init : function(){
												var _this = this;
												$(this.eleContent+'.'+this.clsActive).each(function(){_this.action('#'+$(this).attr('id'))});
												this.event($(this.eleButton));
											},
											event : function($this){
												var _this = this;
												$this.not('.is-clicked').on('click', function(){
													_this.action($(this).attr('href')); return false;
												}).attr('.is-clicked');
											},
											action : function(id){
												$(this.eleButton+'[href="'+id+'"]').attr({'aria-selected':'true', 'aria-expanded':'true'}).parent().addClass(this.clsActive).siblings().removeClass(this.clsActive).children().attr({'aria-selected':'false', 'aria-expanded':'false'});
												$(id).addClass(this.clsActive).attr('aria-hidden', 'false').siblings().removeClass(this.clsActive).attr('aria-hidden', 'true');
											},
										},
									}
