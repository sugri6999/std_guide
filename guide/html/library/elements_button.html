HTML CSS 




									<!-- Block -->
									<button type="button" class="btn btn-type1 btn-block"><span>Block Button</span></button>

									<!-- Float -->
									<div class="btn-wrap">
										<div class="btn-left">
											<button type="button" class="btn btn-type1"><span>btn-left</span></button>
											<button type="button" class="btn btn-type1"><span>btn-left</span></button>
										</div>
										<div class="btn-right">
											<button type="button" class="btn btn-type1"><span>btn-right</span></button>
											<button type="button" class="btn btn-type1"><span>btn-right</span></button>
										</div>
									</div>

									<!-- Flexible Default -->
									<div class="btn-wrap">
										<div class="btn-flexible">
											<span class="item"><button type="button" class="btn btn-type1 btn-block"><span>Flexible Default</span></button></span>
											<span class="item"><button type="button" class="btn btn-type1 btn-block btn-primary"><span>Flexible Default</span></button></span>
											<span class="item"><button type="button" class="btn btn-type1 btn-block"><span>Flexible Default</span></button></span>
										</div>
									</div>

									<!-- Flexible Gutter -->
									<div class="btn-wrap">
										<div class="btn-flexible has-gutter">
											<span class="item"><button type="button" class="btn btn-type1 btn-block"><span>Flexible Gutter</span></button></span>
											<span class="item"><button type="button" class="btn btn-type1 btn-block btn-primary"><span>Flexible Gutter</span></button></span>
											<span class="item"><button type="button" class="btn btn-type1 btn-block"><span>Flexible Gutter</span></button></span>
										</div>
									</div>
								
									/* Common - Layout */
									.btn-wrap::after {content:""; display:block; clear:both;}
									.btn-wrap .btn-left {display:inline-block; float:left;}
									.btn-wrap .btn-right {display:inline-block; float:right;}
									.btn-block {width:100%;}
									.btn-flexible {display:table; width:100%;}
									.btn-flexible .item {display:table-cell; width:auto;}
									.btn-flexible.has-gutter {width:calc(100% + 10px); margin:0 -5px;}
									.btn-flexible.has-gutter .item {padding:0 5px;}

									<button type="button" class="btn btn-type1 btn-bigger"><span>.btn-bigger</span></button>
									<button type="button" class="btn btn-type1 btn-xl"><span>.btn-xl</span></button>
									<button type="button" class="btn btn-type1 btn-lg"><span>.btn-lg</span></button>
									<button type="button" class="btn btn-type1 btn-md"><span>.btn-md</span></button>
									<button type="button" class="btn btn-type1 btn-sm"><span>.btn-sm</span></button>
									<button type="button" class="btn btn-type1 btn-xs"><span>.btn-xl</span></button>
									<button type="button" class="btn btn-type1 btn-thin"><span>.btn-thin</span></button>
								
									/* Common - Size */
									.btn.btn-bigger {min-height:58px; line-height:58px; padding:0 30px; font-size:24px;}
									.btn.btn-xl {min-height:44px; line-height:44px; padding:0 20px; font-size:20px;}
									.btn.btn-lg {min-height:38px; line-height:38px; padding:0 20px; font-size:18px;}
									.btn.btn-md {min-height:34px; line-height:34px; padding:0 10px; font-size:14px;}
									.btn.btn-sm {min-height:28px; line-height:28px; padding:0 10px; font-size:13px;}
									.btn.btn-xs {min-height:22px; line-height:22px; padding:0 10px; font-size:12px;}
									.btn.btn-thin {min-height:18px; line-height:18px; padding:0 10px; font-size:11px;}
  • 여러줄의 텍스트를 고려하는 경우 min-height, line-height, padding 으로 설계한다.
